  def encontrar
  s = "["
  for i in 0...matriz.size
    for j in 0...matriz[0].size
       if (yield(matriz[i][j]))  # Comprobamos que el cuadrado de esa posicion no sea maor que el parametro e
         s += "[#{i},#{j}]"
       end
    end
 end
 s += "]"
 s
end
